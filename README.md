# at-coder-js

use JS for AtCoder

# Before using

npm i

pip3 install --user online-judge-tools
oj login https://beta.atcoder.jp/
npm install -g atcoder-cli
acc login
acc session

# setup

PROBLEM_URL='https://atcoder.jp/contests/agc001/tasks/agc001_a' PROBLEM_NAME='agc001_a' npm run setup

# compile

npm run build

# test

npm run test

# submit

npm run submit
